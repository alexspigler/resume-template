% Custom commands (Section Specific Formatting)

%-------------------------------------------------------------
% 1) EDUCATION
%    Tunable Settings: left column starting point + width,
%                      gap, right column margin, row spacing,
%                      wrapped line spacing
%-------------------------------------------------------------

% Education subheading (2-row table: org + dates / italics row)
\newcommand{\EduSubheading}[4]{%
  \vspace{-2pt}\item
  \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
    \textbf{#1} & #2 \\
    \textit{\fontsize{10.5}{12} #3} & \textit{\small #4} \\
  \end{tabular*}\vspace{-9pt}
}
% Education subheading (3-row variant)
\newcommand{\EduSubheadingExtra}[6]{%
  \vspace{-2pt}\item
  \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
    \textbf{#1} & #2 \\
    \textit{\fontsize{10.5}{12}\selectfont #3} & \textit{\small #4} \\
    \textit{\fontsize{10.5}{12}\selectfont #5} & \textit{\small #6} \\
  \end{tabular*}\vspace{-9pt}
}

% Coursework "label : values" grid
\newlength{\EduLeftStart}    \setlength{\EduLeftStart}{0in}                 % left column starting point
\newlength{\EduLeftWidth}    \setlength{\EduLeftWidth}{0.4\linewidth}       % left column width
\newlength{\EduGap}          \setlength{\EduGap}{0em}                       % gap between both columns
\newlength{\EduRightMargin}  \setlength{\EduRightMargin}{0.08\linewidth}    % right column margin
\newcommand{\EduRowHeight}{\renewcommand{\arraystretch}{1.5}}               % row spacing
\newlength{\EduWrapHeight} \setlength{\EduWrapHeight}{11pt}                 % wrapped line spacing


% Coursework environment
\newenvironment{EduCoursework}{%
  \EduRowHeight
  \begin{tabular}{@{\hspace*{\EduLeftStart}}%
    >{\bfseries}p{\EduLeftWidth}%
    @{\hspace{\EduGap}}%
    >{\setlength{\baselineskip}{\EduWrapHeight}\raggedright\arraybackslash}p{\dimexpr\linewidth-\EduLeftWidth-\EduGap-\EduRightMargin\relax}%
    @{}}
}{%
  \end{tabular}
}


%-------------------------------------------------------------------------
% 2) PROJECTS
%    Tunable Settings: bullet starting point, right margin,
%.                     bullet spacing, wrapped line spacing
%-------------------------------------------------------------------------
\newlength{\ProjBulStart}    \setlength{\ProjBulStart}{0in}                   % bullet starting point
\newlength{\ProjRightMargin}  \setlength{\ProjRightMargin}{0.08\linewidth}    % right margin
\newcommand{\ProjBulHeight}{\renewcommand{\arraystretch}{1.25}}                % bullet spacing
\newlength{\ProjWrapHeight} \setlength{\ProjWrapHeight}{11pt}                 % wrapped line spacing


% Project heading (title left, dates right)
\newcommand{\ProjHeading}[2]{%
  \vspace{0pt}
  \item
  \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
    \small #1 & #2 \\
  \end{tabular*}\vspace{-5pt}
}


%--------------------------------------------------------------
% 3) SKILLS
%    Tunable Settings: left column starting point + width,
%                      gap, right column margin, row spacing,
%                      wrapped line spacing
%--------------------------------------------------------------
\newlength{\SkillLeftStart}    \setlength{\SkillLeftStart}{0in}                  % left column starting point
\newlength{\SkillLeftWidth}    \setlength{\SkillLeftWidth}{0.35\linewidth}       % left column width
\newlength{\SkillGap}          \setlength{\SkillGap}{0.1em}                      % gap between both columns
\newlength{\SkillRightMargin}  \setlength{\SkillRightMargin}{0.05\linewidth}     % right column margin
\newcommand{\SkillRowHeight}{\renewcommand{\arraystretch}{1.2}}                  % row spacing
\newlength{\SkillWrapHeight} \setlength{\SkillWrapHeight}{11pt}                  % wrapped line spacing

% Skills environment
\newenvironment{SkillsTable}{%
  \SkillRowHeight
  \begin{tabular}{@{\hspace*{\SkillLeftStart}}%
    >{\bfseries}p{\SkillLeftWidth}%
    @{\hspace{\SkillGap}}%
    >{\setlength{\baselineskip}{\SkillWrapHeight}\raggedright\arraybackslash}p{\dimexpr\linewidth-\SkillLeftWidth-\SkillGap-\SkillRightMargin-\SkillLeftStart\relax}%
    @{}}
}{%
  \end{tabular}
}


%--------------------------------------------------------------------------
% 4) EXPERIENCE

%                      bullet spacing, wrapped line spacing
%--------------------------------------------------------------------------
\newlength{\ExpBulStart}      \setlength{\ExpBulStart}{0in}                 % bullet starting point
\newlength{\ExpRightMargin}   \setlength{\ExpRightMargin}{0.08\linewidth}   % right margin
\newcommand{\ExpBulHeight}{\renewcommand{\arraystretch}{1}}                 % bullet spacing
\newlength{\ExpWrapHeight} \setlength{\ExpWrapHeight}{11pt}                 % wrapped line spacing

\newcommand{\ExpSubheading}[4]{%
  \vspace{-4pt}\item
    \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \vspace{-3pt}
      \textbf{#1} & #2 \\
      \textit{\small #3} & \textit{\small #4} \\
    \end{tabular*}\vspace{-6pt}
}
